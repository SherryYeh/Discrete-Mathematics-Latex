\begin{algorithm}[H]
  \algsetup{linenosize=\tiny}
  \scriptsize
\begin{algorithmic}[1]
\FOR{$i : = 1$ to $m$}
\STATE {$hit(b_i) := 0$}
\ENDFOR
\STATE {$one\_one := \mathbf{true}$}
\FOR{$j : = 1$ to $n$}
\IF{$hit(f(a_j)) = 0$}
\STATE {$hit( f(a_j) ) := 1$ }
\ELSE
\STATE{$one\_one := \mathbf{false}$}
\ENDIF
\ENDFOR
\RETURN $one\_one$
\end{algorithmic}
\caption{\footnotesize \\ procedure $one\_one$($f$ : function,$a_1,a_2,...,a_n,b_1,b_2,...,b_m$): integers)}
\label{alg:seq}
\end{algorithm}
